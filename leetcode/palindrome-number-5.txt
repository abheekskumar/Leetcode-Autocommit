class Solution(object):
    def isPalindrome(self, x):
        """
        :type x: int
        :rtype: bool
        """
        strNum = str(x)

    def recurPalindrome(self,s):
        if len(s) == 0 or len(s) == 1:
            return True
        if s[0] != s[-1]:
            return False



        return self.recurPalindrome(s[1:-1])
        return self.recurPalindrome(strNum)
