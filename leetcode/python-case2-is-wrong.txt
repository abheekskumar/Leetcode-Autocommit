class Solution(object):
    def removeElement(self, nums, val):
        """
        :type nums: List[int]
        :type val: int
        :rtype: int
        """
        count = {}
        for num in nums:
            count[num] = 1 + count.get(num,0)
        res = 0
        for k,v in count.items():
            if k == val:
                for _ in range(v):
                    nums.remove(k)
            else:
            
                res += v
        return res
